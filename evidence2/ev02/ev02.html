virus <- c("SARS-CoV-2", "MERS-CoV", "SARS-CoV", "HCoV-OC43", "HCoV-229E", "HCoV-NL63", "HCoV-HKU1", "RaTG13", "BtKY72", "WIV1", "PDCoV", "HCoV-HKU9", "HCoV-EMC", "HKU4-1", "HKU5-1", "HKU8-1", "HCoV-229E-related", "HCoV-NL63-related", "HCoV-OC43-related", "HKU1-related")
codigos_acceso <- c("MN908947", "NC_019843", "NC_004718", "NC_006213", "NC_002645", "NC_005831", "NC_006577", "MN996532", "MK211376", "MG772934", "KT446369", "EF065505", "JX869059", "FJ376622", "FJ376624", "NC_006577", "EF065505", "EF065505", "EF065505", "EF065505")

fasta_urls <- paste0("https://www.ncbi.nlm.nih.gov/sviewer/viewer.fcgi?id=", codigos_acceso, "&db=nuccore&report=fasta")

secuencias <- readDNAStringSet(fasta_urls, format = "fasta")
longitudes <- width(secuencias)

barplot(longitudes, names.arg = virus, ylab = "Longitud de secuencia (nmero de bases)", xlab = "Virus", col = "skyblue")

min_length <- min(longitudes)
secuencias_recortadas <- narrow(secuencias, start = 1, width = min_length)

secuencias_dna_bin <- as.DNAbin(secuencias_recortadas)

distancias <- dist.dna(secuencias_dna_bin, model = "JC")

arbol_filogenetico <- njs(distancias)

plot(arbol_filogenetico, main = "rbol filogentico de especies de coronavirus", cex = 0.8, ignore.branch.length = TRUE)
